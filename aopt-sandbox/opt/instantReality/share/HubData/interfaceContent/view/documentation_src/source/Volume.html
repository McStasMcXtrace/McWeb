webvis = webvis || {};

/**
 * @class webvis.Volume
 * @author Timo Sturm <timo.sturm@igd.fraunhofer.de>
 * @constructor
 * Creates a new Volume instance.
 */
webvis.Volume = function( params ) {

    //Check Parameters
    params = webvis.utils.CheckParameters( params, [
        { key: "min", required: false, default: new webvis.Vector3() },
        { key: "max", required: false, default: new webvis.Vector3() }
    ]);

    /**
     * Holds the minimal values of the volume
     * @type {webvis.Vector3}
     * @private
     */
    this._min = params.min;

    /**
     * Holds the maximal values of the volume
     * @type {webvis.Vector3}
     * @private
     */
    this._max = params.max;

    /**
     * Holds the center of the volume
     * @type {webvis.Vector3}
     * @private
     */
    this._center = this._min.Add( this._max ).Scale( 0.5 );

    /**
     * Holds the size of the volume
     * @type {webvis.Vector3}
     * @private
     */
    this._size = this._max.Sub( this._min );
};
