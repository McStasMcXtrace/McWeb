webvis = webvis || {};

/**
 * @class webvis.MetaData
 * @author Timo Sturm <timo.sturm@igd.fraunhofer.de>
 * @constructor
 * Creates a new MetaData instance.
 */
webvis.MetaData = function( params ) {
//----------------------------------------------------------------------------------------------------------------------
// Private Properties
//----------------------------------------------------------------------------------------------------------------------

    /**
     * @property {String}
     * Holds the key of the meta data.
     * @private
     */
    this._key = params.key || null;

    /**
     * @property {String}
     * Holds the value of the meta data.
     * @private
     */
    this._value = params.value || null;

//----------------------------------------------------------------------------------------------------------------------
// Private Methods
//----------------------------------------------------------------------------------------------------------------------

};

//----------------------------------------------------------------------------------------------------------------------
// Extends
//----------------------------------------------------------------------------------------------------------------------

//----------------------------------------------------------------------------------------------------------------------
// Public Properties
//----------------------------------------------------------------------------------------------------------------------

/**
 * @property Key
 * Get or set the key.
 * @type {String}
 */
Object.defineProperty( webvis.MetaData.prototype, "Key", {
    get: function () {
        return this._key;
    },
    set: function ( val ) {
        this._key = val;
    }
});

/**
 * @property Value
 * Get or set the key.
 * @type {String}
 */
Object.defineProperty( webvis.MetaData.prototype, "Value", {
    get: function () {
        return this._value;
    },
    set: function ( val ) {
        this._value = val;
    }
});

//----------------------------------------------------------------------------------------------------------------------
// Public Methods
//----------------------------------------------------------------------------------------------------------------------
