webvis = webvis || {};

/**
 * @class webvis.Event
 * @param {Object} params
 * @param {String} params.type
 * @param {webvis.EventTarget} [params.target=null]
 * @param {webvis.EventTarget} [params.currentTarget=null]
 * @param {Boolean} [params.bubbles=true]
 * @param {Boolean} [params.cancelable=true]
 * @constructor
 */
webvis.Event = function( params ) {

    //Check Parameters
    params = webvis.utils.CheckParameters( params, [
        { key: "type", required: false, default: null },
        { key: "target", required: false, default: null },
        { key: "currentTarget", required: false, default: null },
        { key: "bubbles", required: false, default: false },
        { key: "cancelable", required: false, default: true },
        { key: "data", required: false, default: {} }
    ]);

    /**
     * The type of the event
     * @type {String}
     * @private
     */
    this._type = params.type;

    /**
     * Used to indicate the EventTarget to which the event was originally dispatched.
     * @type {webvis.EventTarget}
     * @private
     */
    this._target = params.target;

    /**
     * Holds an object with user data that can be attached to the event
     * @type {Object}
     * @private
     */
    this._data = params.data;

    /**
     * Used to indicate the EventTarget whose EventListeners are currently being processed. This is particularly
     * useful during capturing and bubbling.
     * @type {webvis.EventTarget}
     * @private
     */
    this._currentTarget = params.currentTarget;

    /**
     * Used to indicate which phase of event flow is currently being evaluated.
     * @type {webvis.EventPhase}
     * @private
     */
    this._eventPhase = webvis.EventPhase.CAPTURING_PHASE;

    /**
     * Used to indicate whether or not an event is a bubbling event. If the event can bubble the value is true,
     * else the value is false.
     * @type {Boolean}
     * @private
     */
    this._bubbles = params.bubbles;


    /**
     * Used to indicate whether or not an event can have its default action prevented. If the default action can be
     * prevented the value is true, else the value is false.
     * @type {Boolean}
     * @private
     */
    this._cancelable = params.cancelable;
};

Object.defineProperty( webvis.Event.prototype, "Type", {
    get: function () {
        return this._type;

        console.error( "[Event] Type is read only");

/**
 * @method
 * The StopPropagation method is used prevent further propagation of an event during event flow. If this method is
 * called by any EventListener the event will cease propagating through the tree. The event will complete dispatch
 * to all listeners on the current EventTarget before event flow stops. This method may be used during any stage
 * of event flow.
 */
webvis.Event.prototype.StopPropagation = function()
{
    if ( this._cancelable )
    {
        this._bubbles = false;
    }
};
