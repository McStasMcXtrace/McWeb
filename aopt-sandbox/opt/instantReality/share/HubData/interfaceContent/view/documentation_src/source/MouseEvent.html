webvis = webvis || {};

/**
 * @class webvis.MouseEvent
 * @param {Object} params
 * @param {String} params.type
 * @param {webvis.EventTarget} [params.target=null]
 * @param {webvis.EventTarget} [params.currentTarget=null]
 * @param {Boolean} [params.bubbles=true]
 * @param {Boolean} [params.cancelable=true]
 * @constructor
 */
webvis.MouseEvent = function( params ) {

    //Check Parameters
    params = webvis.utils.CheckParameters( params, [
        { key: "button", required: false, default: 0 },
        { key: "mouseX", required: false, default: 0 },
        { key: "mouseY", required: false, default: 0 },
        { key: "worldX", required: false, default: 0 },
        { key: "worldY", required: false, default: 0 },
        { key: "worldZ", required: false, default: 0 },
        { key: "normalX", required: false, default: 0 },
        { key: "normalY", required: false, default: 0 },
        { key: "normalZ", required: false, default: 0 }
    ]);

    //Call super-constructor
    webvis.Event.call( this, params );

    /**
     *
     * @type {Number}
     * @private
     */
    this._button = params.button || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._mouseX = params.mouseX || 0;

    /**
     * .
     * @type {Number}
     * @private
     */
    this._mouseY = params.mouseY || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._worldX = params.worldX || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._worldY = params.worldY || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._worldZ = params.worldZ || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._normalX = params.normalX || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._normalY = params.normalY || 0;

    /**
     *
     * @type {Number}
     * @private
     */
    this._normalZ = params.normalZ || 0;

};
