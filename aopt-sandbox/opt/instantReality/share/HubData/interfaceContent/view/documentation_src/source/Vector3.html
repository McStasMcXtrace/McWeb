webvis = webvis || {};

/**
 * @class webvis.Vector3
 * @author Timo Sturm <timo.sturm@igd.fraunhofer.de>
 * @param {Number} x The x-component of the vector
 * @param {Number} y The y-component of the vector
 * @param {Number} z The z-component of the vector
 * @constructor
 * Creates a new Vector3 instance.
 */
webvis.Vector3 = function( x, y, z ) {

    /**
     * Holds the x-component of the vector
     * @type {Number}
     * @private
     */
    this._x = x || 0;

    /**
     * Holds the y-component of the vector
     * @type {Number}
     * @private
     */
    this._y = y || 0;

    /**
     * Holds the z-component of the vector
     * @type {Number}
     * @private
     */
    this._z = z || 0;
};
//----------------------------------------------------------------------------------------------------------------------
// Constructor
//----------------------------------------------------------------------------------------------------------------------

//----------------------------------------------------------------------------------------------------------------------
// Public Properties
//----------------------------------------------------------------------------------------------------------------------

/**
 * @property X
 * Get or set x-component of the vector.
 * @type {Object}
 */
Object.defineProperty( webvis.Vector3.prototype, "X", {
    get: function () {
        return this._x;
    },
    set: function ( val ) {
        this._x = val;
    }
});

/**
 * @property Y
 * Get or set y-component of the vector.
 * @type {Object}
 */
Object.defineProperty( webvis.Vector3.prototype, "Y", {
    get: function () {
        return this._y;
    },
    set: function ( val ) {
        this._y = val;
    }
});

/**
 * @property Z
 * Get or set z-component of the vector..
 * @type {Object}
 */
Object.defineProperty( webvis.Vector3.prototype, "Z", {
    get: function () {
        return this._z;
    },
    set: function ( val ) {
        this._z = val;
    }
});

//----------------------------------------------------------------------------------------------------------------------
// Public Methods
//----------------------------------------------------------------------------------------------------------------------

/**
 * @method Add
 * Add an vector to this vector.
 * @param {webvis.Vector3} that
 * @return {webvis.Vector3}
 */
webvis.Vector3.prototype.Add = function ( that )
{
    return new webvis.Vector3( this._x - that.X, this._y - that.Y, this._z - that.Z );
};

/**
 * @method Sub
 * Subtract an vector to this vector.
 * @param {webvis.Vector3} that
 * @return {webvis.Vector3}
 */
webvis.Vector3.prototype.Sub = function ( that )
{
    return new webvis.Vector3( this._x + that.X, this._y + that.Y, this._z + that.Z );
};

/**
 * @method Scale
 * Scale the vector by the given factor.
 * @param {Number} factor
 * @return {webvis.Vector3}
 */
webvis.Vector3.prototype.Scale = function ( factor )
{
    return new webvis.Vector3( this._x * factor, this._y * factor, this._z * factor );
};


/**
 * @method asArray
 * Returns the vector's content as a JS array
 * @return {Array}
 */
webvis.Vector3.prototype.asArray = function()
{
    return [this._x, this._y, this._z];
};
